3.3 Перевищення температури масла та обмоток над температурою навколишнього середовища

$$ \Theta_{\text{бв}} = \left( 1.05(P_к + P_х) + 2.8 P_{\text{вбр}} \cdot 2.5 \Pi_{\text{кб}} \right)^{0.8} = 22.111\ ^\circ\text{C} \tag{3.30} $$

$$ \Theta_{\text{мб}} = 0.165 k_д \left( \frac{1.05(P_к + P_х)}{n_p \Pi_0 H_б P_б} + \Pi_{\text{ккр}} \right)^{0.6} = 3.31\ ^\circ\text{C} \tag{3.31} $$

$$ \Theta_{\text{мп}} = \Theta_{\text{бв}} + \Theta_{\text{мб}} = 25.421\ ^\circ\text{C} \tag{3.32} $$

$$ \Theta_{\text{мвп}} = 1.2 \cdot \Theta_{\text{мп}} = 30.505\ ^\circ\text{C} < 60\ ^\circ\text{C} \tag{3.33} $$

$$ \Theta_{{оп1}} = \Theta_{{мп}} + \Theta_{{ом1}} + \Theta_1 = 43.8 \, ^\circ C < 65 \, ^\circ C \tag{3.34} $$

$$ \Theta_{{оп2}} = \Theta_{{мп}} + \Theta_{{ом2}} + \Theta_2 = 52.5 \, ^\circ C < 65 \, ^\circ C \tag{3.35} $$

$$ \Theta_{{оп3}} = \Theta_{{мп}} + \Theta_{{ом3}} + \Theta_3 = 33.7 \, ^\circ C < 65 \, ^\circ C \tag{3.36} $$

3.4 Знаходження приблизної маси конструкційних матеріалів та масла трансформатора

$$ G_к = 0.15(G_{{д1}} + G_{{д2}} + G_{{д3}}) = 130.32 \, \text{кг} \tag{3.37} $$

$$ G_{{ач}} = 1.2(G_{{д1}} + G_{{д2}}) + G_{{д3}} + G_к + G_{{в1}} + 10 G_{{в2}} + G_{{ст}} = 4529.13 \, \text{кг} \tag{3.38} $$

$$ G'_{{стр}} = 16 \cdot 2^2 \cdot \gamma_{{ст}} \cdot \delta_{cp} \cdot l_п \cdot h_п = 233.114 \, \text{кг} \tag{3.39} $$

$$ G_{\text{бкр}} = \gamma_{\text{ст}} \left[ 4 \Pi_{\text{ккр}}(b_{\text{кр}} + b_д) + H_б P_б b_{\text{гл}} \right] = 309.33 \, \text{кг} \tag{3.40} $$

$$ V_б = (A - B)(B + \frac{\pi B^2}{4})H_б = 4.04 \, \text{м}^3 \tag{3.41} $$

$$ V_{а} = \frac{G_{\text{ач}}}{\gamma_{\text{ач}}} = 0.81 \, \text{м}^3 \tag{3.42} $$

$$ G_{{мр}} = [2n_р(0.016 b_п h_п) + 16(1.04 \cdot 0.09^2)] \gamma_м = 85.303 \, \text{кг} \tag{3.43} $$

$$ G_p = 0.095(V_б - V_а)\gamma_м = 275.974 \, \text{кг} \tag{3.44} $$

$$ G_м = 1.1 \gamma_м(V_б - V_а) + G_{{мр}} + G_p = 3592.89 \, \text{кг} \tag{3.45} $$

$$ G_т = G_{{ач}} + G_{{бкр}} + G_м + G_H + 1.05 \cdot 3 G_{{yнн}} + 1.25 \cdot 3 G_{{yвн}} + G_{{шк}} + 4G_к = 8804.427 \, \text{кг} \tag{3.46} $$
